function validateAndBook() {
    const name = document.getElementById("name").value.trim();
    const address = document.getElementById("address").value.trim();
    const mobile = document.getElementById("mobile").value.trim();
    const email = document.getElementById("email").value.trim();
    const pickUp = document.getElementById("pick-up").value;
    const destination = document.getElementById("destination").value;
    const date = document.getElementById("cruise-date").value;
    const cabinType = document.getElementById("cabin-type").value;
    const people = document.getElementById("number-of-people").value.trim();
    const paymentMode = document.getElementById("payment-mode").value;

    const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    const mobilePattern = /^[0-9]{10}$/;

    let isValid = true;

    function validateField(field, errorElement, message) {
        if (!field) {
            document.getElementById(errorElement).innerText = message;
            isValid = false;
        } else {
            document.getElementById(errorElement).innerText = "";
        }
    }

    validateField(name, "nameError", "Name is required");
    validateField(address, "addressError", "Address is required");
    validateField(date, "dateError", "Select a date");
    validateField(people, "peopleError", "Enter the number of people");

    if (!mobilePattern.test(mobile)) {
        document.getElementById("mobileError").innerText = "Enter a valid 10-digit number";
        isValid = false;
    } else {
        document.getElementById("mobileError").innerText = "";
    }

    if (!emailPattern.test(email)) {
        document.getElementById("emailError").innerText = "Enter a valid email";
        isValid = false;
    } else {
        document.getElementById("emailError").innerText = "";
    }

    // **New Validation: Only 1 person can book at a time**
    if (people !== "1") {
        document.getElementById("peopleError").innerText = "Only 1 person can book at a time.";
        isValid = false;
    } else {
        document.getElementById("peopleError").innerText = "";
    }

    if (isValid) {
        document.getElementById("form").style.display = "none";
        displayUserDetails(name, address, mobile, email, pickUp, destination, date, cabinType, people, paymentMode);
    }
}
function displayUserDetails(name, address, mobile, email,pickUp, destination, date, cabinType, people, paymentMode) {
    document.getElementById("details-container").innerHTML = `
        <h2>Booking Confirmed!</h2>
        <p><strong>Name:</strong> ${name}</p>
        <p><strong>Address:</strong> ${address}</p>
        <p><strong>Mobile:</strong> ${mobile}</p>
        <p><strong>Email:</strong> ${email}</p>
        <p><strong>Departure Date:</strong> ${date}</p>
        <p><strong>Cabin Type:</strong> ${cabinType}</p>
        <p><strong>Number of People:</strong> ${people}</p>
		<p><strong>Pick-up Point:</strong> ${pickUp}</p>
        <p><strong>Destination:</strong> ${destination}</p>
        <p><strong>Payment Mode:</strong> ${paymentMode}</p>
        <button class="ok-button" onclick="goBack()">OK</button>
    `;

}

        function filterResults() {
            let searchDestination = document.getElementById("searchDestination").value.toLowerCase();
            let searchBudget = document.getElementById("searchBudget").value;
            let searchDuration = document.getElementById("searchDuration").value;

            let cards = document.querySelectorAll(".destination-card");

            cards.forEach(card => {
                let destination = card.getAttribute("data-destination").toLowerCase();
                let price = card.getAttribute("data-price");
                let duration = card.getAttribute("data-duration");

                let matchesDestination = !searchDestination || destination.includes(searchDestination);
                let matchesBudget = !searchBudget || parseInt(price) <= parseInt(searchBudget);
                let matchesDuration = !searchDuration || duration === searchDuration;

                if (matchesDestination && matchesBudget && matchesDuration) {
                    card.style.display = "flex";
                } else {
                    card.style.display = "none";
                }
            });
        }
    </script>
